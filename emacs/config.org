* Emacs

** Layout
#+begin_src emacs-lisp
  ;; Disable tool bar
  (tool-bar-mode -1)

  ;; Disable menu bar
  (menu-bar-mode -1)

  ;; Disable scroll bar
  (scroll-bar-mode -1)

  ;; Disable start up emacs message and go to buffer
  (setq inhibit-startup-message t)

  ;; Disable backup file
  (setq make-backup-file nil)

  ;; Disable auto save
  (setq auto-save-default nil)

  ;; Improve scrolling
  (setq scroll-conservatively 100)

#+end_src
